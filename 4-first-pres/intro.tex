\begin{frame}
    If an edge $uv$ is labelled with a $+$, we denote the edge by $uv^{+}$.
    \begin{definition}
        For any vertex $u \in V(G)$
        \begin{enumerate}
            \item The positive degree of the vertex is
            \begin{equation*}
                d^{+}(u) = |\{v \in V(G) \mid uv^+ \in E(G)\}|
            \end{equation*}
            \item The negative degree of the vertex is
            \begin{equation*}
                d^{-}(u) = |\{v \in V(G) \mid uv^- \in E(G)\}|
            \end{equation*}
            \item The signed degree of a vertex is
            \begin{equation*}
                sd(u) = d^+(u) - d^-(u)
            \end{equation*}
        \end{enumerate}
    \end{definition}
\end{frame}
